  if(!cCanvas||!cBoard||!cBoard.offsetWidth)return;

  const cCtx=cCanvas.getContext('2d');

  function resizeCircuit(){

    if(!cBoard||!cBoard.offsetWidth)return;

    cCanvas.width=cBoard.offsetWidth;

    cCanvas.height=cBoard.offsetHeight||520;

  }

  resizeCircuit();

  if(!cCanvas.width)return;

  window.addEventListener('resize',resizeCircuit);


  const paths=[

    [{x:.05,y:.2},{x:.15,y:.2},{x:.15,y:.5},{x:.35,y:.5}],

    [{x:.35,y:.15},{x:.5,y:.15},{x:.5,y:.35},{x:.65,y:.35}],

    [{x:.2,y:.1},{x:.4,y:.1},{x:.4,y:.3}],

  ];

  const pulses=[];

  paths.forEach((path,pi)=>setTimeout(()=>pulses.push({path,t:Math.random(),speed:0.002+Math.random()*.002,ai:pi<2}),pi*600));


  function drawCircuit(){

    if(!cRunning||!cCanvas.width||!cCanvas.height)return;

    cCtx.clearRect(0,0,cCanvas.width,cCanvas.height);

    const CW=cCanvas.width,CH=cCanvas.height;

    paths.forEach((path,pi)=>{

      cCtx.beginPath();

      path.forEach((pt,i)=>i===0?cCtx.moveTo(pt.x*CW,pt.y*CH):cCtx.lineTo(pt.x*CW,pt.y*CH));

      cCtx.strokeStyle=pi<2?'rgba(245,158,11,.12)':'rgba(0,255,231,.08)';cCtx.lineWidth=1;cCtx.stroke();

      path.forEach(pt=>{cCtx.beginPath();cCtx.arc(pt.x*CW,pt.y*CH,2.5,0,Math.PI*2);cCtx.fillStyle=pi<2?'rgba(245,158,11,.25)':'rgba(0,255,231,.2)';cCtx.fill();});

    });

    pulses.forEach(pulse=>{

      pulse.t+=pulse.speed;if(pulse.t>1)pulse.t=0;

      const path=pulse.path;

      let total=0;const segs=[];

      for(let i=1;i<path.length;i++){const dx=(path[i].x-path[i-1].x)*CW,dy=(path[i].y-path[i-1].y)*CH;segs.push(Math.sqrt(dx*dx+dy*dy));total+=segs[segs.length-1];}

      let rem=pulse.t*total,seg=0;

      for(let i=0;i<segs.length;i++){if(rem<=segs[i]){seg=i;break;}rem-=segs[i];}

      const p0=path[seg],p1=path[Math.min(seg+1,path.length-1)];

      const frac=segs[seg]>0?rem/segs[seg]:0;

      const px=(p0.x+(p1.x-p0.x)*frac)*CW,py=(p0.y+(p1.y-p0.y)*frac)*CH;

      const col=pulse.ai?'245,158,11':'0,255,231';

      const grad=cCtx.createRadialGradient(px,py,0,px,py,12);

      grad.addColorStop(0,`rgba(${col},.9)`);grad.addColorStop(.4,`rgba(${col},.3)`);grad.addColorStop(1,`rgba(${col},0)`);

      cCtx.beginPath();cCtx.arc(px,py,12,0,Math.PI*2);cCtx.fillStyle=grad;cCtx.fill();

      cCtx.beginPath();cCtx.arc(px,py,3,0,Math.PI*2);cCtx.fillStyle=`rgba(${col},1)`;cCtx.fill();

    });

    requestAnimationFrame(drawCircuit);

  }

  drawCircuit();

}
